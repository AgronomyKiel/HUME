unit Umain;

interface

uses
  Windows, Messages, SysUtils, Classes, Graphics, Controls, Forms, Dialogs, UMod, UState;

Type

Tmain = class(TSubmodel)

private

protected

public
  F1 : TVar;   // 


  // Constant Variables


  // State Variables
  C1 : TState;   // Edukt [g/m2]
  C2 : TState;   // Produkt [g/m2]

             // Parameters

             // External Variables
  rgr : TExternV;   // 


  procedure createAll; override; 
  procedure Init(var GlobMod: TMod); override; 
  procedure CalcRates; override; 


published
  Property Var_F1 : TVar read F1 write F1;

  Property St_C1 : TState read C1 write C1;
  Property St_C2 : TState read C2 write C2;


         // Parameters

         // Properties External Variables
  Property Ex_rgr : TExternV read rgr write rgr;


end;  // SubmodelName

procedure Register;

implementation

procedure Tmain.createAll;

begin
  inherited createAll;
  VarCreate('F1', '',0, true, F1, '');  


  StateCreate('C1', '',0, true,C1, 'Edukt [g/m2]');
  StateCreate('C2', '',0, true,C2, 'Produkt [g/m2]');


  // Parameters

         // External Variable
  ExternVCreate('rgr', '',statefield, rgr);
end;


procedure Tmain.init(var GlobMod: TMod);

begin
  inherited init(GlobMod);
  C1.v :=  100;
  C2.v :=  0.1;


end;


procedure Tmain.CalcRates;

begin

   F1.v :=  rgr.v * C1.v*C2;


   C1.c :=  -F1;
   C2.c :=  +F1;


end;



end;

procedure Register;
begin
  RegisterComponents('Ceres Wheat', [Tmain]);
end;

end.
